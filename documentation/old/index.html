JSV

Javascript View - A CoffeeScript template engine

cjsv/menu/main.cjsv 

#main
  .menu_item
    a[href=http://www.google.com] = google

cjsv/menu/sub.cjsv 

(elements)
#sub_menu
  @ for element in elements
    @ +load menu.sub_item element

cjsv/menu/sub_item.cjsv 

(element)
.sub_menu_item
  a[href=+element.href+] = +element.name+
          
@Views =
  menu :
    main : () ->
      _outstream = "
                     <div id="main">
                       <div class="menu_item">
                         <a href="http://www.google.com"> google
                         </a>
                       </div>
                     </div>"
      return _outstream
    sub : (elements) ->
      _outstream = "
                     <div id="sub_menu">"
      for element in elements
        _outstream += Views.menu.sub_item element
      _outstream += "</div>"
      return _outstream
    sub_item : (element) ->
      _outstream = "
                     <div class="sub_menu_item">
                       <a href=""+element.href+""> "+element.name+"
                       </a>
                     </div>"
      return _outstream
